 36 		msg = " > "+msg;

 39 	}

 45 /**
 46  *	@class Manipulate a filepath.
 47  */
 48 FilePath = function(absPath, separator) {
 49 	this.slash =  separator || "/"; 
 50 	this.root = this.slash;
 51 	this.path = [];
 52 	this.file = "";
 53 	
 54 	var parts = absPath.split(/[\\\/]/);
 55 	if (parts) {
 56 		if (parts.length) this.root = parts.shift() + this.slash;
 57 		if (parts.length) this.file =  parts.pop()
 58 		if (parts.length) this.path = parts;
 59 	}
 60 	
 61 	this.path = this.resolvePath();
 62 }
 63 
 64 /** Collapse any dot-dot or dot items in a filepath. */
 65 FilePath.prototype.resolvePath = function() {
 66 	var resolvedPath = [];
 67 	for (var i = 0; i < this.path.length; i++) {
 68 		if (this.path[i] == "..") resolvedPath.pop();
 69 		else if (this.path[i] != ".") resolvedPath.push(this.path[i]);
 70 	}
 71 	return resolvedPath;
 72 }
 73 
 74 /** Trim off the filename. */
 75 FilePath.prototype.toDir = function() {
 76 	if (this.file) this.file = "";
 77 	return this;
 78 }
 79 
 80 /** Go up a directory. */
 81 FilePath.prototype.upDir = function() {
 82 	this.toDir();
 83 	if (this.path.length) this.path.pop();
 84 	return this;
 85 }
 86 
 87 FilePath.prototype.toString = function() {
 88 	return this.root
 89 		+ this.path.join(this.slash)
 90 		+ ((this.path.length > 0)? this.slash : "")
 91 		+ this.file;
 92 }

 94 /**
 95  * Turn a path into just the name of the file.
 96  */
 97 FilePath.fileName = function(path) {
 98 	var nameStart = Math.max(path.lastIndexOf("/")+1, path.lastIndexOf("\\")+1, 0);
 99 	return path.substring(nameStart);
100 }
101 
102 /**
103  * Get the extension of a filename
104  */
105 FilePath.fileExtension = function(filename) {
106    return filename.split(".").pop().toLowerCase();
107 };

109 /**
110  * Turn a path into just the directory part.
111  */
112 FilePath.dir = function(path) {
113 	var nameStart = Math.max(path.lastIndexOf("/")+1, path.lastIndexOf("\\")+1, 0);
114 	return path.substring(0, nameStart-1);
115 }

150 	/**

166 	/**
167 	 * @type string
168 	 */
169 	readFile: function(/**string*/ path) {
170 		if (!IO.exists(path)) {
171 			throw new Error("File doesn't exist there: "+path);
172 		}
173 		return nodejs.fs.readFileSync(path, IO.encoding);
174 	},
175 
176 	/**
177 	 * @param inFile 
178 	 * @param outDir
179 	 * @param [fileName=The original filename]
180 	 */
181 	copyFile: function(/**string*/ inFile, /**string*/ outDir, /**string*/ fileName) {
182 		if (fileName == null) fileName = FilePath.fileName(inFile);
183 	
184 		var inFile = nodejs.fs.openSync(inFile, 'r');
185 		var outFile = nodejs.fs.openSync(IO.join(outDir, fileName), 'w');
186 		
187 		var buffer = new Buffer(4096);
188 		var read;
189 		
190 		while ((read = nodejs.fs.readSync(inFile, buffer, 0, 4096)) != 0) {
191 			nodejs.fs.writeSync(outFile, buffer, 0, read);
192 		}
193 		
194 		nodejs.fs.close(inFile);
195 		nodejs.fs.close(outFile);
196 	},
197 
198 	/**
199 	 * Creates a series of nested directories.
200 	 */
201 	mkPath: function(/**Array*/ path) {
202 		if (path.constructor != Array) path = path.split(/[\\\/]/);
203 		var make = "";
204 		for (var i = 0, l = path.length; i < l; i++) {
205 			make += path[i] + SYS.slash;
206 			if (!IO.exists(make)) {
207 				IO.makeDir(make);
208 			}
209 		}
210 	},
211 	
212 	/**
213 	 * Creates a directory at the given path.
214 	 */
215 	makeDir: function(/**string*/ path) {
216 		if (!IO.exists(path))
217 			nodejs.fs.mkdirSync(path);
218 	},
219 
220 	/**
221 	 * @type string[]
222 	 * @param dir The starting directory to look in.
223 	 * @param [recurse=1] How many levels deep to scan.
224 	 * @returns An array of all the paths to files in the given dir.
225 	 */
226 	ls: function(/**string*/ dir, /**number*/ recurse, _allFiles, _path) {
227 		if (_path === undefined) { // initially
228 			var _allFiles = [];
229 			var _path = [dir];
230 		}
231 		if (_path.length == 0) return _allFiles;
232 		if (recurse === undefined) recurse = 1;
233 		
234 		var dirStats = nodejs.fs.statSync(dir);
235 		if (!dirStats.isDirectory()) return [dir];
236 		var files = nodejs.fs.readdirSync(dir);
237 		
238 		for (var f = 0; f < files.length; f++) {
239 			var file = String(files[f]);
240 			if (file.match(/^\.[^\.\/\\]/)) continue; // skip dot files
241 			
242 			var stats = nodejs.fs.statSync(IO.join(dir, file));
243 	
244 			if (stats.isDirectory()) { // it's a directory
245 				_path.push(file);
246 				if (_path.length-1 < recurse) 
247 					IO.ls(_path.join(SYS.slash), recurse, _allFiles, _path);
248 				_path.pop();
249 			}
250 			else if (stats.isFile()) {
251 				_allFiles.push((_path.join(SYS.slash)+SYS.slash+file).replace(SYS.slash+SYS.slash, SYS.slash));
252 			}
253 		}
254 	
255 		return _allFiles;
256 	},
